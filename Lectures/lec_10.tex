\begin{definition}[Galois group]
    \label{def:galois group}
    \(\Aut(E/F)\) are elements of \(\Gal(E/F)\)
\end{definition}
\begin{example}
    \(\C/\R\) has complex conjugate as it's automorphism  
\end{example}
